# npm
    // 更新npm
    npm install npm@latest -g
    npm -v

    // 查看安装的包
    npm list -g
    // 初始化一个项目
    npm init
    // 安装依赖
    npm install <package_name> 
    --save和--save-dev

    // 卸载
    npm uninstall

    // 检查过时版本
    npm outdated

    // 更新
    npm update <package>
    npm update <package> -g

    // 清空缓存
    npm cache clear
